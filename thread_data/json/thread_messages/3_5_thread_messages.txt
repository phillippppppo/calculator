import Big from "big.js";

function operate(numberOne, numberTwo, operation) {
  const one = Big(numberOne || "0");
  const two = Big(numberTwo || "0");

  switch (operation) {
    case "+":
      return one.plus(two).toString();
    case "-":
      return one.minus(two).toString();
    case "x":
      return one.times(two).toString();
    case "รท":

  }
}

import Big from "big.js";

function calculate(obj, buttonName) {
  if (buttonName === "AC") {
    return {
      total: null,
      next: null,
      operation: null,
    };
  }

  if (isNumber(buttonName)) {
    if (buttonName === "0" && obj.next === "0") {
      return {};
    }

    // If there is an operation, update the next value,
    // otherwise, update the total value
    if (obj.operation) {
      if (obj.next) {
        return { ...obj, next: obj.next + buttonName };
      }
      return { ...obj, next: buttonName };
    }

    if (obj.next) {
      return {
        ...obj,
        next: obj.next === "0" ? buttonName : obj.next + buttonName,
      };
    }
    return {
      ...obj,
      next: buttonName,
    };
  }

  if (buttonName === "%" && obj.operation && obj.next) {
    const result = operate(obj.total, obj.next, obj.operation);
    const percentage = Big(result).div(100).toString();
    return {
      total: percentage,
      next: null,
      operation: null,
    };
  }

  if (buttonName === ".") {
    if (obj.next) {
      if (obj.next.includes(".")) {
        return {};
      }
      return { ...obj, next: obj.next + "." };
    }
    return { ...obj, next: "0." };
  }

  if (buttonName === "+/-") {
    if (obj.next) {
      return { ...obj, next: (-1 * parseFloat(obj.next)).toString() };
    }
    if (obj.total) {
      return { ...obj, total: (-1 * parseFloat(obj.total)).toString() };
    }
    return {};
  }

  if (buttonName === "=") {
    if (obj.next && obj.operation) {
      return {
        total: operate(obj.total, obj.next, obj.operation),
        next: null,
        operation: null,
      };
    }
    // '=' with no operation, do nothing
    return {};
  }

  // If there is an existing operation, calculate the result
  if (obj.operation) {
    return {
      total: operate(obj.total, obj.next, obj.operation),
      next: null,
      operation: buttonName,
    };
  }

  // If there is no operation yet, save the operation and move the next value to total
  if (!obj.next) {
    return { ...obj, operation: buttonName };
  }

  // Save the operation
  return {
    total: obj.next,
    next: null,
    operation: buttonName,
  };
}

function calculate(obj, btnName) {
  let { total, next, operation } = obj;

  if (!isNumber(btnName)) {
    if (btnName === "AC") {
      return { total: null, next: null, operation: null };
    }
    if (btnName === "=") {
      if (total && next && operation) {
          const result = operate(total, next, operation);
          return { total: result, next: null, operation: null };
      }
      throw Error("Invalid calculation");
    }
    if (btnName === "+/-") {
      if (next) {
        next = operate(next, "-1", "x");
        return { total, next, operation };
      }
      throw Error("Invalid input");
    }
    if (btnName === "%") {
      if (next) {
        next = operate(next, "100", "รท");
        return { total, next, operation };
      }
      throw Error("Invalid input");
    }
    if (btnName === ".") {
      if (next) {
        if (!next.includes(".")) {
          next += ".";
          return { total, next, operation };
        }
        throw Error("Invalid input");
      }
      throw Error("Invalid input");
    }
    if (operation) {
      try {
        total = operate(total, next, operation);
      } catch (error) {
        throw Error("Invalid calculation");
      }
    }
    next = null;
    operation = btnName;
    return { total, next, operation };
  }
  
  if (btnName === "0" && next === "0") {
    throw Error("Invalid input");
  }
  
  if (operation && next) {
    total = operate(total, next, operation);
    next = btnName !== "0" ? btnName : null;
    operation = null;
    return { total, next, operation };
  }
  
  if (btnName === "." && !next) {
    next = "0.";
  } else {
    next = next ? next + btnName : btnName;
  }
  
  return { total, next, operation };
}
